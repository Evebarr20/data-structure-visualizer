public class StackExample { //StackExample class
    
    private static final int MAX_SIZE = 100; // Maximum size of the stack
    private int[] arr = new int[MAX_SIZE]; // Creating an integer array to represent the stack
    private int top = -1; // Initializing top pointer to -1, indicating an empty stack

    public boolean isEmpty() { // Method to check if the stack is empty
        return top == -1; // If top is -1, the stack is empty
    }

    public void push(int element) { // Method to push an element onto the stack
        if (top == MAX_SIZE - 1) { // Checking if the stack is full
            System.out.println("Stack overflow"); // Displaying a message if the stack is full
            return; // Exiting the method
        }
        arr[++top] = element; // Incrementing top and adding the element to the stack
    }

    public int pop() { // Method to pop an element from the stack
        if (isEmpty()) { // Checking if the stack is empty
            throw new EmptyStackException(); // Throwing an exception if the stack is empty
        }
        return arr[top--]; // Returning the top element and decrementing top
    }

    public int peek() { // Method to peek at the top element of the stack
        if (isEmpty()) { // Checking if the stack is empty
            throw new EmptyStackException(); // Throwing an exception if the stack is empty
        }
        return arr[top]; // Returning the top element of the stack
    }

    public int getSize() { // Method to get the size of the stack
        return top + 1; // Returning the size of the stack
    }

    public void clear() { // Method to clear the stack
        top = -1; // Resetting top to -1, effectively clearing the stack
    }

    public static void main(String[] args) { // Main method to test the stack implementation
        StackExample stack = new StackExample(); // Creating a new instance of StackExample

        stack.push(1); // Pushing elements onto the stack
        stack.push(2);
        stack.push(3);

        System.out.println("Peek: " + stack.peek()); // Displaying the top element of the stack
        System.out.println("Size: " + stack.getSize()); // Displaying the size of the stack

        stack.pop(); // Popping an element from the stack
        System.out.println("Size after Pop: " + stack.getSize()); // Displaying the size of the stack after popping

        stack.clear(); // Clearing the stack
        System.out.println("Size after clear: " + stack.getSize()); // Displaying the size of the stack after clearing
    }
}

#include

#define MAX_SIZE 100

struct Stack {
    int arr[MAX_SIZE];
    int top;
};

void initialize(struct Stack *stack) {
    stack->top = -1;
}

int isEmpty(struct Stack *stack) {
    return stack->top == -1;
}

void Push(struct Stack *stack, int element) {
    if (stack->top == MAX_SIZE - 1) {
        printf("Stack overflow\n");
        return;
    }
    stack->arr[++stack->top] = element;
}

int Pop(struct Stack *stack) {
    if (isEmpty(stack)) {
        printf("Stack underflow\n");
        exit(1);
    }
    return stack->arr[stack->top--];
}

int Peek(struct Stack *stack) {
    if (isEmpty(stack)) {
        printf("Stack is empty\n");
        exit(1);
    }
    return stack->arr[stack->top];
}

int getSize(struct Stack *stack) {
    return stack->top + 1;
}

void clear(struct Stack *stack) {
    stack->top = -1;
}

int main() {
    struct Stack stack;
    initialize(&stack);

    Push(&stack, 1);
    Push(&stack, 2);
    Push(&stack, 3);

    printf("Peek: %d\n", Peek(&stack));
    printf("Size: %d\n", getSize(&stack));

    Pop(&stack);
    printf("Size after Pop: %d\n", getSize(&stack));

    clear(&stack);
    printf("Size after clear: %d\n", getSize(&stack));

    return 0;
}
